import { Disclosure } from '@lib/components/Disclosure';
import { Stack } from '@lib/components/Layout/Stack';

## Usage

```jsx
import { Disclosure } from '@dodo/ui/components';

<Disclosure label="Disclosure 1" id="disclosure-1">
	<p>Content</p>
</Disclosure>
```

## Default

<div class="not-content">
	<Stack>
		<Disclosure label="Disclosure 1" id="disclosure-1">
			<p>Content</p>
		</Disclosure>
	</Stack>
</div>

## Different label for active state
You can pass a different `labelActive` prop for the active state of the disclosure.

<div class="not-content">
	<Stack>
		<Disclosure label="Open disclosure 2" labelActive="Close disclosure 2" id="disclosure-2">
			<p>Content</p>
		</Disclosure>
	</Stack>
</div>


```jsx
<Disclosure label="Open disclosure 2" labelActive="Close disclosure 2" id="disclosure-2">
	<p>Content</p>
</Disclosure>
```

## Heading level & style
You can pass the `headingLevel` prop to the `Disclosure` component to change the heading level of the disclosure.
Alternatively you can pass the `headingStyledAs` prop to the `Disclosure` component to change the heading style of the disclosure.

<div class="not-content">
	<Stack>
		<Disclosure label="Disclosure 3" headingLevel={3} headingStyledAs={4} id="disclosure-3">
			<p>Content</p>
		</Disclosure>
	</Stack>
</div>


```jsx
<Disclosure label="Disclosure 3" headingLevel={3} headingStyledAs={4} id="disclosure-3">
	<p>Content</p>
</Disclosure>
```

## Accordeon
You can pass the `accordeon` prop to the `Disclosure` component to make it visually an accordeon.

<div class="not-content">
	<Stack>
		<Disclosure label="Disclosure 4" id="disclosure-4" accordeon>
			<p>Content</p>
		</Disclosure>
	</Stack>
</div>

```jsx
<Disclosure label="Disclosure 4" id="disclosure-4" accordeon>
	<p>Content</p>
</Disclosure>
```
