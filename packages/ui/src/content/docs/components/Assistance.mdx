import { Assistance } from '@lib/components/Assistance';

The `Assistance` component is used on all pages with information. It is intended to help a user with questions or problems.

```jsx
import { Assistance } from '@mvoc/ui/components';

<Assistance
	title="Kom je er niet uit?"
	text="Bel ons. We helpen je graag."
	image={{src: 'https://placehold.co/150x150', alt: 'Placeholder'}}
	phone={{
		phoneNumber: '1400',
		openingHourLabel: 'Op werkdagen bereikbaar tussen 08:00 - 20:00 uur',
		openingHour: 8,
		closingHour: 20,
		nowOpen: 'Nu bereikbaar'
	}}
	getHelp={{
		title: 'Staat jouw vraag er niet tussen?'
	}}
	feedback="#"
/>
```
