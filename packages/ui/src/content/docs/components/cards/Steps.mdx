import { Steps, Card } from '@lib/components/Card';
import { Heading } from '@lib/elements/Heading';

The steps component is used to display a series of cards in a vertical row. It is used to display a series of steps in a process, eg. a quarantine plan.

```jsx
import { Card, Steps } from '@mvoc/ui/components';

<Steps>
	<Card>//content </Card>
	<Card>//content </Card>
</Steps>
```
