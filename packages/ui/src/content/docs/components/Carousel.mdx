import { Carousel, CarouselPane } from '@lib/components/Carousel';

The `Carousel` component is used to display a series of images in a horizontal row. Each pane can be a link to another page or a [modal](/components/modal).

## Usage

```jsx
import { Carousel, CarouselPane } from '@dodo/ui/components';

<Carousel>
	<CarouselPane
		id="1"
		headline="Carousel pane 1"
		suffix="Open modal"
		src="https://placehold.co/288x520"
	>
		<p>Modal content</p>
	</CarouselPane>
	<CarouselPane
		id="2"
		headline="Carousel pane 2"
		suffix="Go to link"
		href="#"
		src="https://placehold.co/288x520"
	/>
</Carousel>
```
