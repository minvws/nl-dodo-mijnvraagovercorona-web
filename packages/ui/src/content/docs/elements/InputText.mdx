import { InputText } from '@lib/elements/Input';

## Usage

```jsx
import { InputText } from '@dodo/ui/elements';

<InputText
	type={'search'}
	id={'search-field'}
	placeholder={'Zoek op plaatsnaam...'}
	icon={'looking-glass'}
	clearable
/>
```

## Clearable
You can add a clear button to the input field by providing the `clearable` boolean prop.
When you type something in the input field, the clear button will appear.

<div class="not-content">
	<InputText
		type={'text'}
		id={'text-field'}
		placeholder={'Input Text here'}
		clearable
	/>
</div>

```jsx
<InputText
	type={'text'}
	id={'text-field'}
	placeholder={'Input Text here'}
	clearable
/>
```
