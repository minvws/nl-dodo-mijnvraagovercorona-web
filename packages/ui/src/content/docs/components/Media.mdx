import { Media } from '@lib/components/Media';

The Media component is used to display a image(usually an icon or illustration) to the side of a content block.

## Usage

```jsx
import { Media } from '@dodo/ui/components';

<Media image={{src: 'https://placehold.co/44x44'}}>
	<p>Content</p>
</Media>
```
